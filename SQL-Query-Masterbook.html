01 / SELECT
SELECT — Advanced Patterns
Beyond basic reads: filtering mastery, projections, conditional logic, pagination, and sampling.

Conditional Column Logic CASE WHEN

Conditional expressions
-- CASE as computed column
SELECT
  id,
  email,
  CASE
    WHEN plan = 'enterprise' THEN 'ENT'
    WHEN plan = 'pro'        THEN 'PRO'
    ELSE                          'FREE'
  END                        AS plan_code,

  -- Inline conditional aggregate (very useful)
  SUM(CASE WHEN status = 'paid' THEN amount ELSE 0 END) AS paid_total,
  COUNT(CASE WHEN status = 'failed' THEN 1 END)       AS failed_count,

  -- FILTER (cleaner syntax for PG12+)
  COUNT(*) FILTER (WHERE status = 'active')          AS active_count,
  SUM(amount) FILTER (WHERE created_at >= '2024-01-01') AS ytd_revenue,

  -- NULLIF / COALESCE / GREATEST
  COALESCE(phone, mobile, 'N/A')                       AS contact,
  NULLIF(discount, 0)                                  AS discount_if_any,
  GREATEST(score_a, score_b, score_c)                   AS best_score

FROM users
WHERE deleted_at IS NULL;

Keyset Pagination PRODUCTION PATTERN

OFFSET-based pagination degrades O(n). Keyset (cursor) pagination is O(1) and the only correct choice for large datasets.

Keyset vs OFFSET
-- ❌ NEVER in production for large tables (scans all prior rows)
SELECT * FROM orders ORDER BY created_at DESC
LIMIT 20 OFFSET 100000;   -- reads 100,020 rows to return 20!

-- ✅ Keyset / cursor pagination (requires index on sort key)
-- Page 1: no cursor
SELECT id, created_at, amount FROM orders
WHERE deleted_at IS NULL
ORDER BY created_at DESC, id DESC
LIMIT 20;

-- Next page: pass last row's (created_at, id) as cursor
SELECT id, created_at, amount FROM orders
WHERE deleted_at IS NULL
  AND (created_at, id) < ($1, $2)  -- row-value comparison
ORDER BY created_at DESC, id DESC
LIMIT 20;
-- Requires: CREATE INDEX ON orders (created_at DESC, id DESC)

Sampling & Random Selection

Efficient sampling
-- ❌ Slow: scans entire table and sorts random
SELECT * FROM users ORDER BY RANDOM() LIMIT 100;

-- ✅ TABLESAMPLE — block-level, very fast (approximate)
SELECT * FROM users TABLESAMPLE BERNOULLI(1);  -- ~1% random rows
SELECT * FROM users TABLESAMPLE SYSTEM(0.1);   -- ~0.1% by block

-- ✅ Fast exact-N random sample (for smallish N)
SELECT * FROM users
WHERE id IN (
  SELECT id FROM users
  ORDER BY RANDOM() LIMIT 500
);

-- ✅ Reservoir pattern for consistent random (reproducible)
SELECT * FROM users
WHERE (hashtext(id::text) & 2147483647) < 214748364  -- ~10% deterministic
LIMIT 1000;

DISTINCT ON — Per-Group Latest Row

Latest record per group
-- Get the latest order per user (PostgreSQL DISTINCT ON)
SELECT DISTINCT ON (user_id)
  user_id, id AS order_id, status, amount, created_at
FROM orders
ORDER BY user_id, created_at DESC;  -- DISTINCT ON col must be leftmost in ORDER BY

-- Equivalent with ROW_NUMBER (more portable, sometimes faster)
SELECT user_id, order_id, status, amount, created_at
FROM (
  SELECT *,
    ROW_NUMBER() OVER (PARTITION BY user_id ORDER BY created_at DESC) AS rn
  FROM orders
) ranked
WHERE rn = 1;
02 / INSERT
INSERT — Bulk & Upsert
Single-row, bulk, from-select, returning values, and copy-based mass loading.

INSERT Variants

All insert forms
-- Single row
INSERT INTO users (email, plan, created_at)
VALUES ('alice@co.com', 'pro', NOW())
RETURNING id, created_at;  -- get generated values back

-- Multi-row (one statement = one WAL record = faster)
INSERT INTO users (email, plan) VALUES
  ('b@co.com', 'free'),
  ('c@co.com', 'pro'),
  ('d@co.com', 'enterprise')
RETURNING *;

-- INSERT ... SELECT (from another table / query)
INSERT INTO archived_orders (id, user_id, amount, created_at)
SELECT id, user_id, amount, created_at
FROM   orders
WHERE  created_at < NOW() - interval '2 years'
  AND  status = 'completed';

-- Generate test data (useful for load testing)
INSERT INTO events (user_id, event_type, payload, occurred_at)
SELECT
  (RANDOM() * 10000)::INT + 1               AS user_id,
  (ARRAY['click','view','purchase','logout'])[CEIL(RANDOM()*4)::INT] AS event_type,
  jsonb_build_object('page', CHR(FLOOR(RANDOM()*26)::INT+65))    AS payload,
  NOW() - (RANDOM() * interval '90 days')   AS occurred_at
FROM generate_series(1, 1000000) s;  -- 1M rows

COPY — Fastest Bulk Load FASTEST

COPY for mass data loading
-- Load from CSV file (server-side path)
COPY orders (id, user_id, amount, status, created_at)
FROM '/data/orders_2024.csv'
WITH (FORMAT csv, HEADER true, DELIMITER ',', NULL '');

-- \COPY (client-side, via psql) — no superuser required
\copy orders FROM './orders.csv' WITH (FORMAT csv, HEADER);

-- Export to CSV
COPY (SELECT * FROM orders WHERE status='completed')
TO '/tmp/completed_orders.csv'
WITH (FORMAT csv, HEADER true);

-- Binary format (2-3x faster than CSV for large datasets)
COPY orders FROM '/data/orders.bin' WITH (FORMAT binary);
Performance tip: For bulk inserts, disable indexes and constraints, COPY data, then rebuild. ALTER TABLE orders DISABLE TRIGGER ALL; before and re-enable after. For alloydb, wrap in a single transaction for WAL efficiency.
03 / UPDATE
UPDATE — Safe & Conditional
From-clause joins, conditional updates, safe batch updates, and RETURNING for audit trails.

UPDATE ... FROM (join-based update)

Join-based UPDATE
-- Update orders with user plan from users table
UPDATE orders o
SET
  discount_pct = CASE
    WHEN u.plan = 'enterprise' THEN 20
    WHEN u.plan = 'pro'        THEN 10
    ELSE 0
  END,
  updated_at = NOW()
FROM users u
WHERE o.user_id = u.id
  AND o.discount_pct IS NULL   -- only unset ones
  AND o.status = 'pending'
RETURNING o.id, o.discount_pct, u.email;

Safe Batch UPDATE (avoid table lock)

Chunk-based update — DO NOT update millions at once
-- Update 10M rows safely in batches (avoid WAL surge + lock pile)
DO $$
DECLARE
  updated_count INT;
  batch_size    INT := 10000;
  total         INT := 0;
BEGIN
  LOOP
    WITH batch AS (
      SELECT id FROM orders
      WHERE status = 'pending'
        AND created_at < NOW() - INTERVAL '30 days'
      LIMIT batch_size
      FOR UPDATE SKIP LOCKED   -- skip rows locked by concurrent workers
    )
    UPDATE orders SET
      status = 'expired',
      updated_at = NOW()
    WHERE id IN (SELECT id FROM batch);

    GET DIAGNOSTICS updated_count = ROW_COUNT;
    total := total + updated_count;

    EXIT WHEN updated_count = 0;

    RAISE NOTICE 'Updated % rows (total: %)', updated_count, total;
    PERFORM pg_sleep(0.05);  -- breathing room for other queries
  END LOOP;

  RAISE NOTICE 'Done. Total updated: %', total;
END;
$$;

Optimistic Locking Pattern

Version-based conflict detection
-- Schema: orders has a version INT column
-- App reads row with version=5, sends back version=5 on update
UPDATE orders
SET
  status    = $1,
  version   = version + 1,
  updated_at = NOW()
WHERE id      = $2
  AND version = $3   -- fails silently if another write happened
RETURNING id, version;

-- If RETURNING returns 0 rows → conflict → app must retry
04 / DELETE
DELETE & TRUNCATE
Safe deletions, cascades, soft deletes, and high-speed truncations.

DELETE Patterns

Delete variants
-- Basic with RETURNING (always confirm what was deleted)
DELETE FROM sessions
WHERE expires_at < NOW()
RETURNING id, user_id, expires_at;

-- DELETE via JOIN (using USING clause)
DELETE FROM orders o
USING users u
WHERE o.user_id = u.id
  AND u.status = 'banned'
  AND o.status  = 'pending'
RETURNING o.id, u.email;

-- Soft delete (recommended — preserves audit history)
UPDATE users
SET
  deleted_at    = NOW(),
  deleted_by    = CURRENT_USER,
  email         = email || '_deleted_' || id  -- prevent unique constraint block
WHERE id = $1
  AND deleted_at IS NULL;

-- Batch delete with CTE (safe — preview before deleting)
WITH to_delete AS (
  SELECT id FROM events
  WHERE  occurred_at < NOW() - interval '1 year'
  LIMIT  50000
)
DELETE FROM events
WHERE id IN (SELECT id FROM to_delete)
RETURNING id;

TRUNCATE — Nuclear Option

TRUNCATE variants
-- Instant: resets table to empty (no per-row WAL, no vacuum needed)
TRUNCATE events;

-- Cascade (also empties child tables with FK references)
TRUNCATE users CASCADE;

-- Reset sequence too (auto-increment back to 1)
TRUNCATE orders RESTART IDENTITY CASCADE;

-- Truncate multiple tables in one statement (atomic)
TRUNCATE staging_orders, staging_events, staging_users
RESTART IDENTITY;
TRUNCATE is DDL, not DML — it acquires an ACCESS EXCLUSIVE lock (blocks everything). It IS transactional in PostgreSQL (unlike MySQL), but CANNOT be rolled back after COMMIT. Use DELETE if you need RETURNING or partial deletion.
05 / AGG
GROUP BY & Aggregates
Standard and advanced aggregation — GROUPING SETS, ROLLUP, CUBE, ordered-set and hypothetical aggregates.

Advanced GROUP BY ROLLUP / CUBE

Multi-dimensional aggregation
-- ROLLUP: subtotals + grand total (hierarchical)
SELECT
  DATE_TRUNC('month', created_at) AS month,
  region,
  plan,
  SUM(amount)                      AS revenue,
  COUNT(*)                         AS orders
FROM orders
WHERE status = 'completed'
GROUP BY ROLLUP (
  DATE_TRUNC('month', created_at),
  region,
  plan
)
ORDER BY 1, 2, 3;
-- Produces: (month, region, plan), (month, region), (month), () grand total

-- CUBE: all possible dimension combinations
SELECT region, plan, channel, SUM(revenue)
FROM sales
GROUP BY CUBE (region, plan, channel);  -- 2^3 = 8 groupings

-- GROUPING SETS: explicit combinations
SELECT region, plan, SUM(revenue)
FROM sales
GROUP BY GROUPING SETS (
  (region, plan),   -- by region + plan
  (region),         -- by region only
  ()                -- grand total
);

-- GROUPING() function: detect which columns are subtotal rows
SELECT
  CASE WHEN GROUPING(region) = 1 THEN 'ALL REGIONS' ELSE region END,
  CASE WHEN GROUPING(plan)   = 1 THEN 'ALL PLANS'   ELSE plan   END,
  SUM(revenue)
FROM sales
GROUP BY ROLLUP (region, plan);

Ordered-Set & Percentile Aggregates

Statistical aggregates
SELECT
  plan,
  COUNT(*)                                          AS total_orders,
  ROUND(AVG(amount), 2)                           AS avg_amount,
  PERCENTILE_CONT(0.5) WITHIN GROUP (ORDER BY amount) AS median,
  PERCENTILE_CONT(0.95) WITHIN GROUP (ORDER BY amount) AS p95,
  PERCENTILE_DISC(0.99) WITHIN GROUP (ORDER BY amount) AS p99,
  MODE() WITHIN GROUP (ORDER BY status)            AS most_common_status,
  STDDEV_POP(amount)                                AS std_dev,
  CORR(amount, item_count)                          AS amount_items_correlation
FROM orders
GROUP BY plan;

-- String aggregation
SELECT
  user_id,
  STRING_AGG(tag, ', ' ORDER BY tag)    AS tags_csv,
  ARRAY_AGG(tag ORDER BY created_at)   AS tags_array,
  JSONB_AGG(JSONB_BUILD_OBJECT('id', id, 'tag', tag)) AS tags_json
FROM user_tags
GROUP BY user_id;
06 / WINDOW
Window Functions
The most powerful SQL feature for analytical work. Running totals, rankings, moving averages, lead/lag comparisons.

Ranking Functions

ROW_NUMBER, RANK, DENSE_RANK, NTILE
SELECT
  id, user_id, amount, created_at,

  -- Unique sequential number per partition
  ROW_NUMBER() OVER (PARTITION BY user_id ORDER BY created_at)   AS row_num,

  -- Rank with gaps on tie (1,1,3...)
  RANK() OVER (PARTITION BY user_id ORDER BY amount DESC)       AS rank,

  -- Rank without gaps (1,1,2...)
  DENSE_RANK() OVER (PARTITION BY user_id ORDER BY amount DESC) AS dense_rank,

  -- Bucket into N groups (e.g., quartiles)
  NTILE(4) OVER (ORDER BY amount DESC)                          AS quartile,

  -- Relative position 0.0–1.0
  PERCENT_RANK() OVER (ORDER BY amount)                         AS pct_rank,
  CUME_DIST() OVER (ORDER BY amount)                            AS cume_dist

FROM orders;

LAG / LEAD — Period-over-Period Comparison

MoM growth, previous/next values
WITH monthly AS (
  SELECT
    DATE_TRUNC('month', created_at) AS month,
    SUM(amount)                      AS revenue,
    COUNT(*)                         AS orders
  FROM orders
  WHERE status = 'completed'
  GROUP BY 1
)
SELECT
  month,
  revenue,
  orders,

  -- Previous month's revenue
  LAG(revenue, 1) OVER (ORDER BY month) AS prev_month_rev,

  -- MoM % growth
  ROUND(
    100.0 * (revenue - LAG(revenue, 1) OVER (ORDER BY month))
    / NULLIF(LAG(revenue, 1) OVER (ORDER BY month), 0),
  2) AS mom_growth_pct,

  -- YoY (12 months back)
  LAG(revenue, 12) OVER (ORDER BY month) AS same_month_last_year,

  -- Next month (lookahead)
  LEAD(revenue, 1) OVER (ORDER BY month) AS next_month_rev

FROM monthly
ORDER BY month;

Running Totals & Moving Averages

Frame specification — ROWS BETWEEN
SELECT
  day,
  daily_revenue,

  -- Cumulative total (running sum)
  SUM(daily_revenue)
    OVER (ORDER BY day ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW)
    AS cumulative_revenue,

  -- 7-day moving average
  ROUND(AVG(daily_revenue)
    OVER (ORDER BY day ROWS BETWEEN 6 PRECEDING AND CURRENT ROW), 2)
    AS moving_avg_7d,

  -- 30-day moving sum
  SUM(daily_revenue)
    OVER (ORDER BY day ROWS BETWEEN 29 PRECEDING AND CURRENT ROW)
    AS rolling_30d_revenue,

  -- Running max
  MAX(daily_revenue)
    OVER (ORDER BY day ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW)
    AS all_time_high

FROM daily_revenue_summary
ORDER BY day;

FIRST_VALUE / LAST_VALUE / NTH_VALUE

Value at frame boundary
SELECT
  user_id, order_id, amount, created_at,

  -- First order amount for this user
  FIRST_VALUE(amount)
    OVER (PARTITION BY user_id ORDER BY created_at
          ROWS BETWEEN UNBOUNDED PRECEDING AND UNBOUNDED FOLLOWING)
    AS first_order_amount,

  -- Most recent order amount
  LAST_VALUE(amount)
    OVER (PARTITION BY user_id ORDER BY created_at
          ROWS BETWEEN UNBOUNDED PRECEDING AND UNBOUNDED FOLLOWING)
    AS latest_order_amount,

  -- Second order (useful for cohort analysis)
  NTH_VALUE(amount, 2)
    OVER (PARTITION BY user_id ORDER BY created_at
          ROWS BETWEEN UNBOUNDED PRECEDING AND UNBOUNDED FOLLOWING)
    AS second_order_amount

FROM orders;
07 / CTE
CTEs & Recursive Queries
Readable multi-step queries, data pipelines in SQL, and recursive tree/graph traversal.

Chained CTEs — SQL Data Pipeline

Multi-step analytical pipeline
WITH
-- Step 1: filter valid orders
valid_orders AS (
  SELECT * FROM orders
  WHERE status = 'completed'
    AND created_at >= '2024-01-01'
    AND amount > 0
),
-- Step 2: enrich with user data
enriched AS (
  SELECT
    o.*,
    u.plan,
    u.country,
    u.signup_date
  FROM valid_orders o
  JOIN users u ON u.id = o.user_id
),
-- Step 3: segment users
segmented AS (
  SELECT
    *,
    CASE
      WHEN amount >= 1000 THEN 'whale'
      WHEN amount >= 100  THEN 'mid'
      ELSE                     'small'
    END AS segment
  FROM enriched
),
-- Step 4: final aggregation
summary AS (
  SELECT
    country, plan, segment,
    COUNT(*) AS orders,
    SUM(amount) AS revenue,
    ROUND(AVG(amount), 2) AS avg_order
  FROM segmented
  GROUP BY country, plan, segment
)
SELECT * FROM summary
ORDER BY revenue DESC;

Recursive CTEs — Trees & Graphs

Org chart / category hierarchy traversal
-- Table: employees(id, name, manager_id, department)

-- Full org tree from CEO down
WITH RECURSIVE org_tree AS (
  -- Anchor: root node (no manager)
  SELECT
    id, name, manager_id, department,
    0            AS depth,
    name::TEXT   AS path
  FROM employees
  WHERE manager_id IS NULL

  UNION ALL

  -- Recursive: join children
  SELECT
    e.id, e.name, e.manager_id, e.department,
    t.depth + 1,
    t.path || ' → ' || e.name
  FROM employees e
  JOIN org_tree t ON t.id = e.manager_id
  WHERE t.depth < 10   -- cycle guard
)
SELECT
  REPEAT('  ', depth) || name AS tree_view,
  depth, department, path
FROM org_tree
ORDER BY path;

-- Find all ancestors of an employee (bottom-up)
WITH RECURSIVE ancestors AS (
  SELECT id, name, manager_id, 0 AS level
  FROM employees WHERE id = $1

  UNION ALL

  SELECT e.id, e.name, e.manager_id, a.level + 1
  FROM employees e
  JOIN ancestors a ON a.manager_id = e.id
)
SELECT * FROM ancestors ORDER BY level;

Writeable CTEs — DML in CTEs

INSERT + UPDATE in one statement
-- Archive-and-delete in a single atomic statement
WITH moved AS (
  DELETE FROM orders
  WHERE created_at < NOW() - INTERVAL '2 years'
    AND status = 'completed'
  RETURNING *
)
INSERT INTO orders_archive
SELECT *, NOW() AS archived_at
FROM moved;
-- All or nothing — no partial archive possible
08 / JOIN
JOIN Patterns
Every join type with real use cases, anti-joins, self-joins, lateral joins, and performance patterns.

Join Type Reference

All join types
-- INNER JOIN: only matching rows on both sides
SELECT o.id, u.email, o.amount
FROM orders o INNER JOIN users u ON u.id = o.user_id;

-- LEFT JOIN: all orders, user may be NULL (deleted users)
SELECT o.id, COALESCE(u.email, 'deleted') AS email, o.amount
FROM orders o LEFT JOIN users u ON u.id = o.user_id;

-- ANTI-JOIN: users who have never placed an order
SELECT u.id, u.email
FROM users u
LEFT JOIN orders o ON o.user_id = u.id
WHERE o.id IS NULL;    -- the anti-join trick

-- SELF-JOIN: compare a table to itself
SELECT
  a.id AS emp_id, a.name AS employee,
  b.name AS manager
FROM employees a
LEFT JOIN employees b ON b.id = a.manager_id;

-- CROSS JOIN: cartesian product (use carefully)
SELECT c.name, s.name
FROM colors c CROSS JOIN sizes s;  -- all combinations

LATERAL JOIN — Correlated Subquery as Table

LATERAL — top-N per group with join
-- Top 3 orders per user (efficient with LATERAL)
SELECT u.id, u.email, o.id AS order_id, o.amount
FROM users u
CROSS JOIN LATERAL (
  SELECT id, amount
  FROM   orders
  WHERE  user_id = u.id          -- references outer row
  ORDER BY created_at DESC
  LIMIT  3
) o;

-- LATERAL with function (e.g., unnest, JSON)
SELECT u.id, tag
FROM users u
CROSS JOIN LATERAL UNNEST(u.tags) AS t(tag);

-- LEFT JOIN LATERAL (include users with no orders)
SELECT u.email, COALESCE(o.amount, 0) AS latest_amount
FROM users u
LEFT JOIN LATERAL (
  SELECT amount FROM orders
  WHERE user_id = u.id
  ORDER BY created_at DESC
  LIMIT 1
) o ON TRUE;
09 / SUBQUERY
Subqueries & EXISTS
Correlated subqueries, EXISTS vs IN performance, scalar subqueries, and derived tables.

EXISTS vs IN — Know the Difference

EXISTS (use this), IN (careful with NULLs)
-- IN: fine for small, static lists; NULL in subquery = silent miss
SELECT * FROM users
WHERE id IN (1, 2, 3);

-- ⚠️ IN with subquery + NULLs: if subquery returns any NULL,
--    rows NOT matching return unknown (not false) → rows disappear!
SELECT * FROM orders
WHERE user_id NOT IN (SELECT id FROM users WHERE plan='free');
-- DANGER: if users has NULL ids, this returns 0 rows!

-- ✅ EXISTS: stops on first match, handles NULLs correctly, fast
SELECT * FROM users u
WHERE EXISTS (
  SELECT 1 FROM orders o
  WHERE o.user_id = u.id
    AND o.status = 'completed'
    AND o.amount > 500
);

-- NOT EXISTS (safe anti-join alternative)
SELECT * FROM users u
WHERE NOT EXISTS (
  SELECT 1 FROM orders o
  WHERE o.user_id = u.id
);

Scalar Subqueries & Correlated

Per-row computed values
SELECT
  u.id, u.email,
  -- Scalar subquery: 1 value per user row
  (SELECT COUNT(*) FROM orders o WHERE o.user_id = u.id) AS total_orders,
  (SELECT SUM(amount) FROM orders o WHERE o.user_id = u.id) AS ltv,
  (SELECT MAX(created_at) FROM orders o WHERE o.user_id = u.id) AS last_order_date

FROM users u
WHERE u.plan = 'enterprise';
-- Note: for large tables, prefer LEFT JOIN with GROUP BY over correlated subqueries

-- ANY / ALL operators
SELECT * FROM products
WHERE price > ALL (SELECT price FROM products WHERE category='budget');

SELECT * FROM products
WHERE price > ANY (SELECT price FROM products WHERE category='premium');
10 / SET OPS
Set Operations
UNION, INTERSECT, EXCEPT — combining result sets with dedup or not.

UNION / INTERSECT / EXCEPT

Set operation patterns
-- UNION ALL (faster — no dedup); UNION (deduplicates)
-- Use case: combine two reporting periods
SELECT '2023' AS year, region, SUM(revenue) FROM sales_2023 GROUP BY region
UNION ALL
SELECT '2024' AS year, region, SUM(revenue) FROM sales_2024 GROUP BY region;

-- INTERSECT: users who are both customers AND partners
SELECT email FROM customers
INTERSECT
SELECT email FROM partners;

-- EXCEPT: customers who are NOT partners
SELECT email FROM customers
EXCEPT
SELECT email FROM partners;

-- Multi-source data consolidation with source tag
SELECT 'web' AS source, id, email, created_at FROM web_users
UNION ALL
SELECT 'app',              id, email, created_at FROM app_users
UNION ALL
SELECT 'api',              id, email, created_at FROM api_users
ORDER BY created_at DESC;
11 / JSONB
JSON & JSONB Operations
PostgreSQL's JSONB is a full document store inside a relational database. Master it for flexible schemas.

Extracting JSON Values

JSON operators and path queries
-- -> returns JSONB, ->> returns text
SELECT
  payload -> 'address'                     AS address_json,
  payload ->> 'email'                      AS email_text,
  payload -> 'address' ->> 'city'         AS city,
  payload #> '{address,zip}'               AS zip_json,   -- path array
  payload #>> '{address,zip}'              AS zip_text,
  (payload ->> 'score')::NUMERIC          AS score,       -- cast to type
  payload -> 'tags' -> 0                   AS first_tag    -- array index
FROM events;

-- Filtering on JSON fields
SELECT * FROM events
WHERE  payload ->> 'event_type' = 'purchase'
  AND  (payload ->> 'amount')::NUMERIC > 100
  AND  payload @> '{"status": "completed"}'   -- containment check
  AND  payload ? 'coupon_code';              -- key existence

Modifying JSONB

Non-destructive JSON updates
-- Merge (shallow): overwrite specific keys
UPDATE users
SET metadata = metadata || '{"verified": true, "tier": "gold"}'::jsonb
WHERE id = $1;

-- Deep set a nested key with jsonb_set
UPDATE users
SET metadata = jsonb_set(
  metadata,
  '{address,city}'::text[],    -- path
  '"Mumbai"'::jsonb,            -- new value (must be valid JSON)
  true                         -- create if missing
)
WHERE id = $1;

-- Remove a key
UPDATE users
SET metadata = metadata - 'temp_token'     -- remove key
WHERE metadata ? 'temp_token';

-- Append to a JSON array
UPDATE users
SET metadata = jsonb_set(
  metadata,
  '{tags}':: text[],
  (COALESCE(metadata->'tags', '[]'::jsonb) || '"beta_tester"'::jsonb)
);

Expanding & Aggregating JSON

jsonb_each, jsonb_array_elements, build
-- Expand JSON object to key-value rows
SELECT id, key, value
FROM configs, jsonb_each(settings);

-- Expand JSON array to rows
SELECT id, jsonb_array_elements_text(tags) AS tag
FROM products;

-- Aggregate rows into JSON object
SELECT
  user_id,
  jsonb_object_agg(key, value)   AS settings_map,
  jsonb_agg(
    jsonb_build_object('id', order_id, 'amount', amount)
  )                               AS orders_json
FROM orders
GROUP BY user_id;
12 / ARRAYS
Arrays & Composite Types
Native array columns, unnesting, array operators, and composite row types.

Array Operations

Array DML and query patterns
-- Creating and querying arrays
SELECT
  tags,
  tags[1]                              AS first_tag,    -- 1-indexed!
  array_length(tags, 1)               AS tag_count,
  array_to_string(tags, ', ')         AS tags_csv,
  array_upper(tags, 1)                AS last_index,
  tags[2:4]                            AS slice         -- subarray
FROM products;

-- Array containment and overlap
SELECT * FROM products
WHERE tags @> ARRAY['sale', 'new']::text[]  -- contains all
   OR tags && ARRAY['featured']::text[];       -- overlaps any

-- ANY / ALL with arrays
SELECT * FROM products WHERE 'sale' = ANY(tags);
SELECT * FROM orders   WHERE status = ANY(ARRAY['pending', 'processing']);

-- Append / prepend / remove from array
UPDATE products
SET tags = array_append(tags, 'clearance')
WHERE stock < 10;

UPDATE products
SET tags = array_remove(tags, 'new')
WHERE created_at < NOW() - INTERVAL '30 days';

-- UNNEST: expand array to rows (with ordinality for index)
SELECT p.id, t.tag, t.pos
FROM products p,
UNNEST(p.tags) WITH ORDINALITY AS t(tag, pos);

Transaction Patterns

Isolation, savepoints, advisory locks
-- Explicit transaction with isolation
BEGIN ISOLATION LEVEL REPEATABLE READ;

  SELECT balance FROM accounts WHERE id = 1 FOR UPDATE;  -- lock it
  UPDATE accounts SET balance = balance - 500 WHERE id = 1;
  UPDATE accounts SET balance = balance + 500 WHERE id = 2;

COMMIT;

-- SAVEPOINT: partial rollback within a transaction
BEGIN;
  INSERT INTO orders ... RETURNING id INTO v_order_id;

  SAVEPOINT after_order;

  INSERT INTO payments (order_id, ...) VALUES (v_order_id, ...);
  -- If payment insert fails:
  ROLLBACK TO SAVEPOINT after_order;  -- order intact, payment rolled back

  INSERT INTO payments ... -- retry with corrected data

COMMIT;

-- Advisory locks: application-level mutex (no table lock)
-- Use for: preventing duplicate job processing
SELECT pg_try_advisory_lock(12345);     -- returns true if acquired
SELECT pg_advisory_lock(12345);         -- blocks until acquired
SELECT pg_advisory_unlock(12345);       -- release
-- Session-scoped; released automatically on disconnect
16 / LOCK
Explicit Locking Patterns
SELECT FOR UPDATE, SKIP LOCKED, NOWAIT, and lock-free patterns for queues and concurrent workers.

Row-Level Lock Variants

FOR UPDATE / SKIP LOCKED — queue processing
-- FOR UPDATE: lock rows for update (blocks concurrent updates)
BEGIN;
SELECT id, amount FROM accounts
WHERE id = $1
FOR UPDATE;  -- blocks if another txn holds lock
UPDATE ...
COMMIT;

-- FOR UPDATE NOWAIT: fail immediately if locked (no waiting)
SELECT id FROM accounts WHERE id = $1
FOR UPDATE NOWAIT;  -- raises ERROR if locked → app catches & retries

-- SKIP LOCKED: skip locked rows — perfect for job queues
-- Multiple workers can grab different jobs concurrently
BEGIN;
SELECT id, payload
FROM job_queue
WHERE status = 'pending'
ORDER BY created_at
LIMIT 10
FOR UPDATE SKIP LOCKED;  -- this worker grabs 10 unlocked jobs

UPDATE job_queue SET status = 'processing', worker_id = pg_backend_pid()
WHERE id = ANY(ARRAY[...selected ids...]);
COMMIT;

-- FOR SHARE: allows concurrent reads, blocks updates
SELECT * FROM accounts WHERE id = $1 FOR SHARE;
17 / UPSERT
MERGE / ON CONFLICT
Atomic upsert patterns — the most important DML pattern for idempotent data pipelines.

ON CONFLICT (PostgreSQL Upsert)

INSERT ... ON CONFLICT — all patterns
-- DO NOTHING: ignore if duplicate
INSERT INTO user_events (user_id, event_type, occurred_at)
VALUES ($1, $2, $3)
ON CONFLICT (user_id, event_type, occurred_at) DO NOTHING;

-- DO UPDATE: upsert — update specific fields on conflict
INSERT INTO user_stats (user_id, total_orders, total_revenue, last_updated)
VALUES ($1, $2, $3, NOW())
ON CONFLICT (user_id) DO UPDATE SET
  total_orders   = user_stats.total_orders + EXCLUDED.total_orders,
  total_revenue  = user_stats.total_revenue + EXCLUDED.total_revenue,
  last_updated   = EXCLUDED.last_updated   -- EXCLUDED = the rejected insert row
RETURNING *;

-- Conditional upsert: only update if new value is larger
INSERT INTO leaderboard (user_id, high_score)
VALUES ($1, $2)
ON CONFLICT (user_id) DO UPDATE SET
  high_score = GREATEST(leaderboard.high_score, EXCLUDED.high_score);

-- Conflict on partial unique index
CREATE UNIQUE INDEX idx_active_users_email
  ON users (email) WHERE deleted_at IS NULL;

INSERT INTO users (email, plan)
VALUES ($1, $2)
ON CONFLICT (email) WHERE deleted_at IS NULL  -- matches partial index
DO UPDATE SET plan = EXCLUDED.plan;

Finding Gaps in Sequences

Missing IDs in a sequence
-- Find missing IDs in orders table
SELECT s.id AS missing_id
FROM generate_series(
  (SELECT MIN(id) FROM orders),
  (SELECT MAX(id) FROM orders)
) s(id)
LEFT JOIN orders o ON o.id = s.id
WHERE o.id IS NULL;

-- Gaps with range: show start and end of each gap
WITH gaps AS (
  SELECT
    id,
    LEAD(id) OVER (ORDER BY id) AS next_id
  FROM orders
)
SELECT
  id + 1       AS gap_start,
  next_id - 1  AS gap_end,
  next_id - id - 1 AS gap_size
FROM gaps
WHERE next_id > id + 1
ORDER BY gap_size DESC;

Islands — Grouping Consecutive Events

Session detection, consecutive date groups
-- Group consecutive active days per user (streak detection)
WITH
active_days AS (
  SELECT DISTINCT
    user_id,
    occurred_at::DATE AS day
  FROM events
),
with_prev AS (
  SELECT
    user_id, day,
    day - (ROW_NUMBER() OVER (
      PARTITION BY user_id ORDER BY day
    ) * INTERVAL '1 day')::DATE AS island_id
  FROM active_days
)
SELECT
  user_id,
  MIN(day)  AS streak_start,
  MAX(day)  AS streak_end,
  COUNT(*) AS streak_days
FROM with_prev
GROUP BY user_id, island_id
ORDER BY streak_days DESC;
20 / PERF
Performance Query Patterns
Patterns that scale: avoiding N+1, writing planner-friendly queries, index-only scans, and partial indexes.

Index-Only Scans — Covering Indexes

Cover all needed columns in the index
-- Query: find pending orders with amount
SELECT id, amount FROM orders WHERE status = 'pending';

-- Normal index: still needs heap fetch for amount
CREATE INDEX idx_orders_status ON orders (status);

-- Covering index: amount INCLUDED → pure index-only scan
CREATE INDEX idx_orders_status_cover
  ON orders (status)
  INCLUDE (id, amount);  -- INCLUDE cols not in predicate/order

-- Partial index: only index what you query
CREATE INDEX idx_orders_pending
  ON orders (created_at)
  WHERE status = 'pending';  -- tiny index, only pending rows

-- Verify index-only scan (no heap fetches)
EXPLAIN (ANALYZE, BUFFERS)
  SELECT id, amount FROM orders WHERE status = 'pending';
-- Look for "Index Only Scan" + "Heap Fetches: 0"

Avoiding Implicit Casts & Non-SARGable Predicates

Write index-friendly WHERE clauses
-- ❌ Function on indexed column = full scan
WHERE LOWER(email) = 'alice@co.com'        -- index on email not used
WHERE DATE(created_at) = '2024-01-15'     -- index on created_at not used
WHERE id::TEXT = '12345'                  -- implicit cast = no index

-- ✅ Rewrite to be SARGable
WHERE email = 'alice@co.com'              -- use expression index if needed
WHERE created_at >= '2024-01-15'
  AND created_at < '2024-01-16'           -- range instead of DATE()
WHERE id = 12345                          -- correct type, no cast

-- Expression index to fix LOWER() issue
CREATE INDEX idx_users_email_lower ON users (LOWER(email));

-- ❌ OR prevents index usage sometimes
WHERE status = 'pending' OR status = 'processing'
-- ✅ Use IN or = ANY()
WHERE status = ANY(ARRAY['pending', 'processing'])

  p_plan    TEXT

CREATE OR REPLACE FUNCTION get_user_summary(p_user_id BIGINT)

  amount      NUMERIC,
  status      TEXT,

)

$$
BEGIN
  RETURN QUERY

    FROM  orders
    WHERE user_id = p_user_id
      AND deleted_at IS NULL
    ORDER BY created_at DESC;
END;
$$;

Stored Procedure with Exception Handling

Full procedure with error handling and logging
CREATE OR REPLACE PROCEDURE process_payment(
  p_order_id  BIGINT,
  p_amount    NUMERIC,
  p_method    TEXT
)
LANGUAGE plpgsql AS
$$
DECLARE
  v_user_id   BIGINT;
  v_balance   NUMERIC;
  v_payment_id BIGINT;
BEGIN
  -- Lock order row
  SELECT user_id INTO STRICT v_user_id
  FROM  orders
  WHERE id = p_order_id
  FOR UPDATE;

  -- Validate balance
  SELECT balance INTO v_balance
  FROM  accounts
  WHERE user_id = v_user_id;

  IF v_balance < p_amount THEN
    RAISE EXCEPTION 'Insufficient balance: % available, % required',
      v_balance, p_amount
      USING ERRCODE = 'P0001';
  END IF;

  -- Deduct and record
  UPDATE accounts SET balance = balance - p_amount
  WHERE user_id = v_user_id;

  INSERT INTO payments (order_id, amount, method, status)
  VALUES (p_order_id, p_amount, p_method, 'completed')
  RETURNING id INTO v_payment_id;

  UPDATE orders SET status = 'paid', payment_id = v_payment_id
  WHERE id = p_order_id;

  RAISE NOTICE 'Payment % completed for order %', v_payment_id, p_order_id;

EXCEPTION
  WHEN SQLSTATE 'P0001' THEN
    RAISE;  -- re-raise business logic errors
  WHEN OTHERS THEN
    RAISE EXCEPTION 'Payment failed for order %: %',
      p_order_id, SQLERRM;
END;
$$;

-- Call it
CALL process_payment(1001, 99.99, 'credit_card');

Trigger Function

Audit trail trigger
-- Audit log trigger: records every change to orders
CREATE OR REPLACE FUNCTION audit_orders()
RETURNS TRIGGER LANGUAGE plpgsql AS
$$
BEGIN
  INSERT INTO order_audit (
    order_id, operation, old_status, new_status,
    old_amount, new_amount, changed_by, changed_at
  ) VALUES (
    COALESCE(NEW.id, OLD.id),
    TG_OP,                          -- INSERT / UPDATE / DELETE
    OLD.status, NEW.status,
    OLD.amount, NEW.amount,
    CURRENT_USER,
    NOW()
  );
  RETURN NEW;
END;
$$;

CREATE TRIGGER trg_audit_orders
  AFTER INSERT OR UPDATE OR DELETE ON orders
  FOR EACH ROW EXECUTE FUNCTION audit_orders();

Staging Table ETL Pattern

Load → validate → merge pattern
-- Step 1: Load raw data into staging (no constraints = fast)
CREATE TEMP TABLE staging_orders (LIKE orders INCLUDING DEFAULTS);

\copy staging_orders FROM './new_orders.csv' WITH (FORMAT csv, HEADER);

-- Step 2: Validate (find bad rows)
SELECT * FROM staging_orders
WHERE  amount <= 0
    OR user_id IS NULL
    OR status NOT IN ('pending', 'completed', 'failed');

-- Step 3: Upsert into production
INSERT INTO orders
SELECT * FROM staging_orders
WHERE amount > 0 AND user_id IS NOT NULL
ON CONFLICT (id) DO UPDATE SET
  status     = EXCLUDED.status,
  amount     = EXCLUDED.amount,
  updated_at = NOW();

-- Step 4: Drop staging
DROP TABLE staging_orders;

-- Create partitioned table
CREATE TABLE events (
  id          BIGINT NOT NULL,
  user_id     BIGINT,
  event_type  TEXT,

  occurred_at TIMESTAMPTZ NOT NULL
) PARTITION BY RANGE (occurred_at);

CREATE TABLE events_2024_01 PARTITION OF events
  FOR VALUES FROM ('2024-01-01') TO ('2024-02-01');

CREATE TABLE events_2024_02 PARTITION OF events
  FOR VALUES FROM ('2024-02-01') TO ('2024-03-01');

SELECT * FROM events
WHERE occurred_at BETWEEN '2024-01-01' AND '2024-01-31';

ALTER TABLE events DETACH PARTITION events_2024_01 CONCURRENTLY;

CREATE TABLE events_2025_01 (...);
ALTER TABLE events ATTACH PARTITION events_2025_01
  FOR VALUES FROM ('2025-01-01') TO ('2025-02-01');
