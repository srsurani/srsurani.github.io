01
SELECT Mastery
Beyond basic SELECT — filtering, conditional logic, sampling, lateral joins, and set operations that power real applications.
🔎 Conditional Column Selection (CASE WHEN)essential
▼
Use CASE WHEN for computed columns, data bucketing, status labels, and conditional aggregation in a single pass.
SQL — CASE WHEN patterns
-- Basic CASE: status labels
SELECT
  id,
  order_total,
  CASE
    WHEN order_total >= 10000 THEN 'Enterprise'
    WHEN order_total >= 1000  THEN 'Business'
    WHEN order_total >= 100   THEN 'Standard'
    ELSE 'Small'
  END AS tier,

  -- CASE inside aggregate (conditional count)
  COUNT(CASE WHEN status = 'paid'    THEN 1 END) AS paid_count,
  COUNT(CASE WHEN status = 'pending' THEN 1 END) AS pending_count,
  SUM(CASE WHEN status = 'refunded' THEN order_total ELSE 0 END) AS refunded_amount,

  -- FILTER syntax (cleaner PG alternative)
  COUNT(*) FILTER (WHERE status = 'paid')    AS paid_filter,
  SUM(order_total) FILTER (WHERE status = 'paid') AS paid_total
FROM orders
GROUP BY id, order_total, status;
💡
FILTER (WHERE ...) is cleaner than CASE WHEN ... THEN 1 END for conditional aggregation and runs at the same speed.
🎲 TABLESAMPLE — Fast Random Samplingadvanced
▼
Get statistical samples without scanning the entire table. Critical for ML feature engineering and approximate analytics on billions of rows.
SQL — Sampling methods
-- BERNOULLI: each row has 1% chance of inclusion (true random)
SELECT * FROM events TABLESAMPLE BERNOULLI(1);

-- SYSTEM: page-level sampling (faster, less random, good for large tables)
SELECT * FROM events TABLESAMPLE SYSTEM(0.1); -- ~0.1% of pages

-- Reproducible sample with seed
SELECT * FROM events TABLESAMPLE BERNOULLI(5) REPEATABLE(42);

-- Random N rows (small tables / dev testing)
SELECT * FROM users ORDER BY RANDOM() LIMIT 100;

-- Better: reservoir-style (no full sort for large tables)
SELECT * FROM users WHERE id % 100 = 0 LIMIT 1000; -- systematic
🔗 LATERAL JOIN — Correlated Subquery as Joinadvanced
▼
LATERAL lets a subquery reference columns from preceding FROM items — enables top-N per group, correlated function calls, and dynamic row generation.
SQL — LATERAL patterns
-- Top 3 most recent orders per customer
SELECT c.id, c.name, o.id AS order_id, o.created_at, o.total
FROM customers c
CROSS JOIN LATERAL (
  SELECT id, created_at, total
  FROM orders
  WHERE customer_id = c.id        -- references outer row
  ORDER BY created_at DESC
  LIMIT 3
) o;

-- LATERAL with LEFT JOIN (include customers with no orders)
SELECT c.id, c.name, o.order_id
FROM customers c
LEFT JOIN LATERAL (
  SELECT id AS order_id
  FROM orders WHERE customer_id = c.id
  ORDER BY created_at DESC LIMIT 1
) o ON TRUE;

-- LATERAL calling a set-returning function per row
SELECT u.id, tags.tag
FROM users u,
LATERAL unnest(u.tag_array) AS tags(tag);
⚙️ SET Operations — UNION, INTERSECT, EXCEPTessential
▼
SQL
-- UNION ALL (keep dupes, faster) vs UNION (dedup, slower)
SELECT id, email, 'active' AS source FROM active_users
UNION ALL
SELECT id, email, 'archived' FROM archived_users;

-- INTERSECT: rows in BOTH result sets
SELECT user_id FROM premium_subscribers
INTERSECT
SELECT user_id FROM active_last_30_days;

-- EXCEPT: rows in first set but NOT in second
SELECT user_id FROM all_users
EXCEPT
SELECT user_id FROM users_who_ordered;

-- Combine with CTE for complex set logic
WITH churned AS (
  SELECT user_id FROM users WHERE last_login < now() - interval '90 days'
  EXCEPT
  SELECT user_id FROM active_subscriptions
)
SELECT * FROM churned;
📋 DISTINCT ON — PostgreSQL's Unique Featureadvanced
▼
DISTINCT ON keeps the first row for each distinct group — a powerful PostgreSQL-specific feature for "latest per group" queries without window functions.
SQL — DISTINCT ON
-- Latest order per customer
SELECT DISTINCT ON (customer_id)
  customer_id, id AS order_id, created_at, total
FROM orders
ORDER BY customer_id, created_at DESC;
-- DISTINCT ON column must be first in ORDER BY

-- Latest price per product from price_history
SELECT DISTINCT ON (product_id)
  product_id, price, effective_date
FROM price_history
ORDER BY product_id, effective_date DESC;

-- Most recent event per user per event_type
SELECT DISTINCT ON (user_id, event_type)
  user_id, event_type, event_time, metadata
FROM events
ORDER BY user_id, event_type, event_time DESC;
💡
DISTINCT ON with the right index (on the DISTINCT column + ORDER BY column DESC) can be faster than equivalent ROW_NUMBER() window function patterns.
🏷️ VALUES as Inline Tableadvanced
▼
Use VALUES to create inline tables — great for lookups, category mapping, and test data without temp tables.
SQL
-- Inline lookup table with VALUES
SELECT o.id, o.status, m.label, m.priority
FROM orders o
JOIN (
  VALUES
    ('pending',   'Awaiting Payment', 1),
    ('paid',      'Confirmed',         2),
    ('shipped',   'In Transit',        3),
    ('delivered', 'Complete',          4),
    ('refunded',  'Refunded',          5)
) AS m(status, label, priority) ON o.status = m.status;

-- Generate a series of dates with VALUES
SELECT dt::date
FROM generate_series(
  '2024-01-01'::date,
  '2024-12-31'::date,
  '1 day'::interval
) dt;
🧮 COALESCE, NULLIF, GREATEST, LEASTessential
▼
SQL — Null handling and comparisons
-- COALESCE: first non-null value
SELECT
  COALESCE(preferred_name, first_name, 'Anonymous') AS display_name,

  -- NULLIF: returns null if equal (avoid divide-by-zero)
  total_revenue / NULLIF(order_count, 0) AS avg_order_value,

  -- GREATEST / LEAST: across columns
  GREATEST(q1_sales, q2_sales, q3_sales, q4_sales) AS best_quarter,
  LEAST(q1_sales, q2_sales, q3_sales, q4_sales) AS worst_quarter,

  -- IS DISTINCT FROM (null-safe equality check)
  CASE WHEN old_email IS DISTINCT FROM new_email
    THEN 'email_changed'
    ELSE 'same'
  END AS change_type

FROM users;
🔢 String Formatting & Type Castingessential
▼
SQL — Type handling
-- Safe cast (returns null instead of error)
SELECT CAST('123abc' AS INTEGER);    -- throws error
SELECT NULLIF(REGEXP_REPLACE('123abc', '[^0-9]',''), '')::int; -- safe

-- Number formatting
SELECT
  TO_CHAR(1234567.89, 'FM$999,999,999.00') AS formatted,
  TO_CHAR(now(), 'YYYY-MM-DD HH24:MI:SS') AS ts,
  ROUND(3.14159::numeric, 2)              AS rounded,
  TRUNC(3.99)                              AS truncated;

-- String operations
SELECT
  UPPER(name), LOWER(name), INITCAP(name),
  TRIM('  hello  '),
  LPAD('5', 4, '0'),        -- '0005'
  SPLIT_PART('a,b,c', ',', 2), -- 'b'
  REGEXP_REPLACE(phone, '[^0-9]', '', 'g') AS clean_phone,
  SUBSTR(description, 1, 100) || '...'    AS excerpt
FROM contacts;

INSERT Patterns
Multi-row inserts, INSERT from SELECT, RETURNING clause, conflict handling, and batch strategies.
📥 Multi-Row & INSERT ... SELECTessential
▼
SQL
-- Multi-row insert (single round-trip)
INSERT INTO products (name, sku, price, category_id)
VALUES
  ('Widget A', 'WGT-001', 29.99, 1),
  ('Widget B', 'WGT-002', 49.99, 1),
  ('Gadget X', 'GDG-001', 99.99, 2);

-- INSERT from SELECT (copy / transform data)
INSERT INTO order_archive (id, user_id, total, archived_at)
SELECT id, user_id, total, now()
FROM orders
WHERE created_at < now() - interval '2 years'
  AND status = 'delivered';

-- RETURNING: get generated values back
INSERT INTO users (email, created_at)
VALUES ('user@example.com', now())
RETURNING id, created_at;

-- RETURNING multiple rows into a CTE
WITH new_orders AS (
  INSERT INTO orders (user_id, total)
  SELECT id, 0 FROM new_users
  RETURNING id, user_id
)
INSERT INTO order_items (order_id, product_id, qty)
SELECT id, 1, 1 FROM new_orders;
🧩 INSERT with DEFAULT & Sequence Controlessential
▼
SQL
-- Explicit DEFAULT keyword
INSERT INTO events (type, user_id, occurred_at)
VALUES ('page_view', 42, DEFAULT); -- uses column default (now())

-- Get next sequence value without inserting
SELECT nextval('orders_id_seq');

-- Peek at current without advancing
SELECT currval('orders_id_seq');

-- Reset sequence (after bulk load)
SELECT setval('orders_id_seq', (SELECT MAX(id) FROM orders));

-- UUID primary key (recommended for distributed systems)
INSERT INTO sessions (id, user_id, data)
VALUES (gen_random_uuid(), 42, '{"theme":"dark"}')
RETURNING id;
🔄 INSERT OVERRIDING SYSTEM VALUEadvanced
▼
When migrating data, you need to insert specific IDs into IDENTITY / SERIAL columns without losing auto-increment behavior.
SQL — Identity column override
-- Table with GENERATED ALWAYS AS IDENTITY
CREATE TABLE products (
  id    BIGINT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
  name  TEXT
);

-- Normal insert (system assigns ID)
INSERT INTO products (name) VALUES ('Test') RETURNING id;

-- Force a specific ID (migration use case)
INSERT INTO products (id, name)
OVERRIDING SYSTEM VALUE
VALUES (9999, 'Migrated Product');

-- After forced inserts, sync the sequence
SELECT setval(
  pg_get_serial_sequence('products', 'id'),
  (SELECT MAX(id) FROM products)
);
03
UPDATE Techniques
UPDATE with JOINs, conditional updates, batch updates, UPDATE RETURNING, and safe large-table update strategies.
🔗 UPDATE with FROM (JOIN-style Update)essential
▼
PostgreSQL uses UPDATE ... FROM instead of UPDATE ... JOIN. Essential for updating one table based on data from another.
SQL
-- Update orders with data from customers
UPDATE orders o
SET
  customer_name = c.full_name,
  customer_tier = c.tier,
  updated_at    = now()
FROM customers c
WHERE o.customer_id = c.id
  AND o.customer_name IS DISTINCT FROM c.full_name;

-- Update using subquery in FROM
UPDATE products p
SET total_sold = sub.qty_sum
FROM (
  SELECT product_id, SUM(qty) AS qty_sum
  FROM order_items
  GROUP BY product_id
) sub
WHERE p.id = sub.product_id;

-- UPDATE with RETURNING (get before + after)
UPDATE accounts
SET balance = balance - 100
WHERE id = 42
RETURNING id, balance AS new_balance, balance + 100 AS old_balance;
⚡ Conditional Update (only changed rows)performance
▼
Avoid unnecessary WAL writes and trigger fires by only updating rows where the value actually changed.
SQL
-- IS DISTINCT FROM: null-safe inequality (avoid touching unchanged rows)
UPDATE users
SET
  email     = $1,
  phone     = $2,
  updated_at = now()
WHERE id = $3
  AND (
    email IS DISTINCT FROM $1
    OR phone IS DISTINCT FROM $2
  );

-- Batch conditional update with CASE
UPDATE products
SET
  price = CASE category
    WHEN 'electronics' THEN price * 0.90  -- 10% off
    WHEN 'clothing'    THEN price * 0.85  -- 15% off
    ELSE price * 0.95
  END,
  updated_at = now()
WHERE on_sale = true;

-- Safe large-table update in batches (avoid long lock)
DO $$
DECLARE
  updated_count INT;
BEGIN
  LOOP
    UPDATE events
    SET processed = true
    WHERE id IN (
      SELECT id FROM events
      WHERE processed = false
      LIMIT 10000
    );
    GET DIAGNOSTICS updated_count = ROW_COUNT;
    EXIT WHEN updated_count = 0;
    PERFORM pg_sleep(0.05); -- yield to other queries
  END LOOP;
END$$;
⚠️
Batched updates with sleep are the safe pattern for large tables in production — a single UPDATE of 10M+ rows holds locks and generates massive WAL, causing replication lag and contention.
🔢 Atomic Counter & Toggle Updatesessential
▼
SQL
-- Atomic increment (thread-safe without transactions)
UPDATE posts
SET view_count = view_count + 1
WHERE id = 42
RETURNING view_count;

-- Toggle boolean
UPDATE feature_flags
SET enabled = NOT enabled
WHERE flag_name = 'dark_mode'
RETURNING enabled;

-- Bounded counter (clamp to max 9999)
UPDATE rate_limits
SET tokens = LEAST(tokens + 10, 9999)
WHERE user_id = $1;

-- Decrement with floor at 0 (no negatives)
UPDATE inventory
SET stock = GREATEST(stock - $1, 0)
WHERE product_id = $2
  AND stock >= $1  -- only if enough stock
RETURNING stock;

DELETE & Truncate
Safe delete strategies, CASCADE patterns, soft deletes, and efficient data purging for large tables.
🗑️ DELETE with USING (Join-style Delete)essential
▼
SQL
-- Delete using another table (PostgreSQL USING clause)
DELETE FROM order_items oi
USING orders o
WHERE oi.order_id = o.id
  AND o.status = 'cancelled'
  AND o.created_at < now() - interval '1 year';

-- Delete with RETURNING (audit what was removed)
DELETE FROM sessions
WHERE expires_at < now()
RETURNING id, user_id, created_at;

-- DELETE oldest N rows (retention policy)
DELETE FROM audit_log
WHERE id IN (
  SELECT id FROM audit_log
  ORDER BY created_at ASC
  LIMIT 10000
)
RETURNING *;

-- Soft delete pattern (mark as deleted, never hard delete)
UPDATE users
SET
  deleted_at  = now(),
  deleted_by  = current_user,
  email       = email || '._deleted_' || extract(epoch from now())::text
WHERE id = $1
  AND deleted_at IS NULL
RETURNING id, deleted_at;
⚡ Batched DELETE for Large Tablesperformance
▼
SQL — Safe batch purge
-- Chunked delete: purge events older than 90 days in 5000-row batches
DO $$
DECLARE
  deleted_rows INT;
  total_deleted INT := 0;
  cutoff_date TIMESTAMPTZ := now() - interval '90 days';
BEGIN
  LOOP
    DELETE FROM events
    WHERE id IN (
      SELECT id FROM events
      WHERE created_at < cutoff_date
      LIMIT 5000
      FOR UPDATE SKIP LOCKED  -- skip rows locked by other sessions
    );
    GET DIAGNOSTICS deleted_rows = ROW_COUNT;
    total_deleted := total_deleted + deleted_rows;
    EXIT WHEN deleted_rows = 0;
    COMMIT;                   -- commit each batch
    PERFORM pg_sleep(0.1);
  END LOOP;
  RAISE NOTICE 'Total deleted: %', total_deleted;
END$$;
ℹ️
FOR UPDATE SKIP LOCKED is the pattern for queue processing and batch jobs — it skips rows locked by other transactions instead of waiting, enabling parallel workers.
🧹 TRUNCATE Patternsessential
▼
SQL
-- TRUNCATE: fastest way to empty a table (no row-by-row logging)
TRUNCATE TABLE staging_data;

-- TRUNCATE with CASCADE (also empties FK-dependent tables)
TRUNCATE TABLE orders CASCADE;  -- also truncates order_items etc.

-- TRUNCATE multiple tables atomically
TRUNCATE TABLE staging_users, staging_orders, staging_products;

-- TRUNCATE and reset sequences
TRUNCATE TABLE orders RESTART IDENTITY;

-- TRUNCATE is transactional in PostgreSQL!
BEGIN;
  TRUNCATE TABLE temp_results;
  -- ... do work ...
ROLLBACK;  -- data comes back!
🚨
TRUNCATE takes an ACCESS EXCLUSIVE lock (blocks all reads and writes). For tables with FK references, use CASCADE carefully — it empties child tables too. Always TRUNCATE in transactions for safety.

UPSERT / MERGE Patterns
INSERT ON CONFLICT, conditional upserts, and the SQL MERGE statement — the most-used patterns in data pipelines and sync jobs.
🔄 INSERT ... ON CONFLICT (Full Upsert)essential
▼
SQL — ON CONFLICT patterns
-- DO NOTHING: ignore duplicates
INSERT INTO events (id, type, user_id, occurred_at)
VALUES ($1, $2, $3, now())
ON CONFLICT (id) DO NOTHING;

-- DO UPDATE: update specific columns on conflict
INSERT INTO user_stats (user_id, total_orders, total_spend, last_updated)
VALUES ($1, $2, $3, now())
ON CONFLICT (user_id) DO UPDATE SET
  total_orders = user_stats.total_orders + EXCLUDED.total_orders,
  total_spend  = user_stats.total_spend  + EXCLUDED.total_spend,
  last_updated = EXCLUDED.last_updated;
-- EXCLUDED = the row that would have been inserted

-- Upsert only if newer (timestamp guard)
INSERT INTO product_cache (product_id, data, fetched_at)
VALUES ($1, $2, now())
ON CONFLICT (product_id) DO UPDATE SET
  data      = EXCLUDED.data,
  fetched_at = EXCLUDED.fetched_at
WHERE product_cache.fetched_at < EXCLUDED.fetched_at;
-- WHERE clause: only update if EXCLUDED is newer

-- Conflict on composite unique constraint
INSERT INTO daily_metrics (user_id, metric_date, value)
VALUES ($1, $2, $3)
ON CONFLICT (user_id, metric_date) DO UPDATE SET
  value = GREATEST(daily_metrics.value, EXCLUDED.value); -- keep max
🔀 MERGE Statement (PostgreSQL 15+)expert
▼
MERGE is the SQL standard way to synchronize a target table with a source — insert, update, and delete in a single statement.
SQL — MERGE (PG 15+)
MERGE INTO customers AS target
USING customer_updates AS source
ON target.id = source.id

WHEN MATCHED AND source.action = 'delete' THEN
  DELETE

WHEN MATCHED AND source.action = 'update' THEN
  UPDATE SET
    name       = source.name,
    email      = source.email,
    updated_at = now()

WHEN NOT MATCHED AND source.action = 'insert' THEN
  INSERT (id, name, email, created_at)
  VALUES (source.id, source.name, source.email, now());

-- MERGE for data warehouse ETL pattern
MERGE INTO dim_products t
USING stg_products s ON t.product_key = s.product_key
WHEN MATCHED THEN UPDATE SET
  name = s.name, price = s.price, modified_dt = now()
WHEN NOT MATCHED THEN INSERT
  (product_key, name, price, created_dt)
  VALUES (s.product_key, s.name, s.price, now());

JOIN Patterns
Every JOIN type with real use cases — including the ones developers get wrong most often.
🗺️ All JOIN Types Referenceessential
▼
SQL — JOIN visual guide
-- INNER JOIN: only matching rows in both tables
SELECT o.id, c.name FROM orders o
INNER JOIN customers c ON c.id = o.customer_id;

-- LEFT JOIN: all orders, customer data if exists (NULL otherwise)
SELECT o.id, c.name FROM orders o
LEFT JOIN customers c ON c.id = o.customer_id;

-- LEFT JOIN to find rows WITHOUT a match (anti-join pattern)
SELECT c.* FROM customers c
LEFT JOIN orders o ON o.customer_id = c.id
WHERE o.id IS NULL;  -- customers who never ordered

-- FULL OUTER JOIN: all rows from both sides
SELECT a.id, b.id FROM table_a a
FULL OUTER JOIN table_b b ON a.key = b.key;

-- CROSS JOIN: cartesian product (all combinations)
SELECT s.size, c.color
FROM sizes s CROSS JOIN colors c;

-- SELF JOIN: compare rows within same table
SELECT e.name AS employee, m.name AS manager
FROM employees e
LEFT JOIN employees m ON m.id = e.manager_id;
🎯 NOT EXISTS vs NOT IN vs Anti-Joinperformance
▼
Three ways to find "unmatched" rows — with very different performance characteristics. NOT EXISTS is almost always best.
SQL — Anti-join comparison
-- ❌ NOT IN: BROKEN if subquery can return NULLs
SELECT * FROM customers
WHERE id NOT IN (SELECT customer_id FROM orders);
-- Returns 0 rows if ANY customer_id is NULL in orders!

-- ✅ NOT EXISTS: correct and generally fast
SELECT * FROM customers c
WHERE NOT EXISTS (
  SELECT 1 FROM orders o WHERE o.customer_id = c.id
);

-- ✅ Anti-join with LEFT JOIN (planner may choose hash join)
SELECT c.* FROM customers c
LEFT JOIN orders o ON o.customer_id = c.id
WHERE o.id IS NULL;

-- ✅ EXCEPT (set-based, clean)
SELECT id FROM customers
EXCEPT
SELECT customer_id FROM orders WHERE customer_id IS NOT NULL;
🚨
NOT IN with nullable columns is a silent bug. If the subquery returns even one NULL, the entire NOT IN returns 0 rows. Always use NOT EXISTS or LEFT JOIN IS NULL instead.
🔑 JOIN with Range & Inequality Conditionsadvanced
▼
SQL
-- Non-equi join: assign tier based on range table
SELECT u.id, u.total_spend, t.tier_name
FROM users u
JOIN loyalty_tiers t
  ON u.total_spend BETWEEN t.min_spend AND t.max_spend;

-- Temporal join: find price effective at order time
SELECT o.id, o.created_at, ph.price
FROM orders o
JOIN price_history ph ON
  ph.product_id = o.product_id
  AND o.created_at >= ph.valid_from
  AND o.created_at <  COALESCE(ph.valid_to, '9999-12-31');

-- Nearest-neighbor join using tsrange overlap
SELECT b.*, r.room_name
FROM bookings b
JOIN rooms r ON r.id = b.room_id
WHERE tsrange(b.checkin, b.checkout) &&
      tsrange('2024-12-24', '2024-12-27');

CTEs & Recursive Queries
Common Table Expressions for readable multi-step queries, data pipelines, and recursive tree/graph traversal.
🌀 Multi-Step CTE Pipelineessential
▼
SQL — CTE pipeline
WITH
-- Step 1: Filter active customers from last 90 days
active_customers AS (
  SELECT DISTINCT customer_id
  FROM orders
  WHERE created_at > now() - interval '90 days'
    AND status = 'delivered'
),
-- Step 2: Compute their spend
customer_spend AS (
  SELECT
    o.customer_id,
    COUNT(*) AS order_count,
    SUM(o.total) AS total_spend,
    AVG(o.total) AS avg_order
  FROM orders o
  JOIN active_customers ac USING (customer_id)
  GROUP BY o.customer_id
),
-- Step 3: Rank by spend
ranked AS (
  SELECT *,
    NTILE(4) OVER (ORDER BY total_spend DESC) AS quartile
  FROM customer_spend
)
SELECT
  c.name, r.order_count, r.total_spend, r.avg_order,
  CASE r.quartile
    WHEN 1 THEN 'Top 25%'
    WHEN 2 THEN 'Upper Mid'
    ELSE 'Lower Half'
  END AS segment
FROM ranked r
JOIN customers c ON c.id = r.customer_id
ORDER BY r.total_spend DESC;
🌳 Recursive CTE — Hierarchical Treeexpert
▼
Traverse parent-child hierarchies (org charts, categories, comments, bill of materials) using recursive CTEs.
SQL — Recursive tree traversal
-- Org chart: find all reports under a manager
WITH RECURSIVE org_tree AS (
  -- Anchor: start with the root manager
  SELECT id, name, manager_id, 0 AS depth, name::text AS path
  FROM employees
  WHERE id = 1  -- root node

  UNION ALL

  -- Recursive: join each employee to their parent
  SELECT e.id, e.name, e.manager_id,
    ot.depth + 1,
    ot.path || ' → ' || e.name
  FROM employees e
  INNER JOIN org_tree ot ON e.manager_id = ot.id
  WHERE ot.depth < 10  -- guard against cycles
)
SELECT depth, path, id, name
FROM org_tree
ORDER BY path;

-- Recursive: build full category path
WITH RECURSIVE cat_path AS (
  SELECT id, name, parent_id, name::text AS full_path
  FROM categories WHERE parent_id IS NULL
  UNION ALL
  SELECT c.id, c.name, c.parent_id, cp.full_path || ' / ' || c.name
  FROM categories c
  JOIN cat_path cp ON c.parent_id = cp.id
)
SELECT id, full_path FROM cat_path ORDER BY full_path;

-- Find all ancestors of a node (bottom-up)
WITH RECURSIVE ancestors AS (
  SELECT id, parent_id, name FROM categories WHERE id = 42
  UNION ALL
  SELECT c.id, c.parent_id, c.name
  FROM categories c
  JOIN ancestors a ON c.id = a.parent_id
)
SELECT * FROM ancestors;
✍️ Writeable CTEs (INSERT/UPDATE/DELETE in CTE)expert
▼
SQL — Data-modifying CTEs
-- Move rows atomically: delete + insert in one statement
WITH moved AS (
  DELETE FROM job_queue
  WHERE id = (
    SELECT id FROM job_queue
    WHERE status = 'pending'
    ORDER BY priority DESC, created_at ASC
    LIMIT 1
    FOR UPDATE SKIP LOCKED
  )
  RETURNING *
)
INSERT INTO job_processing (job_id, worker_id, started_at)
SELECT id, pg_backend_pid(), now()
FROM moved;

-- Multi-table insert from single source
WITH new_user AS (

Window Functions
The most powerful SQL feature for analytics — ranking, running totals, lag/lead, percentiles, and moving averages.
🏆 Ranking Functions (ROW_NUMBER, RANK, DENSE_RANK)essential
▼
SQL — Ranking
SELECT
  product_id, category, revenue,

  -- ROW_NUMBER: unique rank, no ties (1,2,3,4...)
  ROW_NUMBER() OVER (PARTITION BY category ORDER BY revenue DESC)
    AS row_num,

  -- RANK: ties get same rank, then gap (1,1,3,4...)
  RANK() OVER (PARTITION BY category ORDER BY revenue DESC)
    AS rank,

  -- DENSE_RANK: ties same rank, no gap (1,1,2,3...)
  DENSE_RANK() OVER (PARTITION BY category ORDER BY revenue DESC)
    AS dense_rank,

  -- NTILE: divide into N buckets
  NTILE(4) OVER (ORDER BY revenue DESC) AS quartile,

  -- PERCENT_RANK: position as 0-1 fraction
  PERCENT_RANK() OVER (ORDER BY revenue) AS pct_rank,

  -- CUME_DIST: cumulative distribution 0-1
  CUME_DIST() OVER (ORDER BY revenue) AS cume_dist

FROM product_sales;

-- Use in WHERE via CTE (can't filter window functions directly)
WITH ranked AS (
  SELECT *, ROW_NUMBER() OVER (PARTITION BY category ORDER BY revenue DESC) rn
  FROM product_sales
)
SELECT * FROM ranked WHERE rn <= 3;  -- top 3 per category
📈 Running Totals, Moving Averages, Cumulative Sumsessential
▼
SQL — Cumulative & moving window
SELECT
  order_date,
  daily_revenue,

  -- Running total (cumulative sum)
  SUM(daily_revenue) OVER (
    ORDER BY order_date
    ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW
  ) AS running_total,

  -- 7-day moving average
  AVG(daily_revenue) OVER (
    ORDER BY order_date
    ROWS BETWEEN 6 PRECEDING AND CURRENT ROW
  ) AS moving_avg_7d,

  -- 30-day rolling sum
  SUM(daily_revenue) OVER (
    ORDER BY order_date
    ROWS BETWEEN 29 PRECEDING AND CURRENT ROW
  ) AS rolling_30d,

  -- Running count
  COUNT(*) OVER (ORDER BY order_date) AS day_number,

  -- Partition by month, running total resets each month
  SUM(daily_revenue) OVER (
    PARTITION BY DATE_TRUNC('month', order_date)
    ORDER BY order_date
  ) AS month_running_total

FROM daily_revenue_stats
ORDER BY order_date;
⏮️ LAG & LEAD — Previous/Next Row Valuesessential
▼
SQL — LAG / LEAD
SELECT
  order_date,
  daily_revenue,

  -- Previous day's revenue
  LAG(daily_revenue, 1, 0) OVER (ORDER BY order_date) AS prev_day,

  -- Day-over-day change
  daily_revenue - LAG(daily_revenue) OVER (ORDER BY order_date)
    AS day_delta,

  -- Day-over-day % change
  ROUND(
    100.0 * (daily_revenue - LAG(daily_revenue) OVER (ORDER BY order_date))
    / NULLIF(LAG(daily_revenue) OVER (ORDER BY order_date), 0), 2
  ) AS pct_change,

  -- Same day last week
  LAG(daily_revenue, 7) OVER (ORDER BY order_date) AS week_ago,

  -- Next day forecast reference
  LEAD(daily_revenue, 1) OVER (ORDER BY order_date) AS next_day

FROM daily_revenue_stats
ORDER BY order_date;
📊 FIRST_VALUE, LAST_VALUE, NTH_VALUEadvanced
▼
SQL
SELECT
  customer_id, order_date, total,

  -- First purchase date
  FIRST_VALUE(order_date) OVER (
    PARTITION BY customer_id ORDER BY order_date
  ) AS first_purchase,

  -- Most recent purchase
  LAST_VALUE(order_date) OVER (
    PARTITION BY customer_id ORDER BY order_date
    ROWS BETWEEN UNBOUNDED PRECEDING AND UNBOUNDED FOLLOWING
  ) AS last_purchase,
  -- LAST_VALUE requires ROWS BETWEEN to extend to full partition!

  -- Second order value
  NTH_VALUE(total, 2) OVER (
    PARTITION BY customer_id ORDER BY order_date
    ROWS BETWEEN UNBOUNDED PRECEDING AND UNBOUNDED FOLLOWING
  ) AS second_order_value

FROM orders;
⚠️
LAST_VALUE's default window frame only goes to CURRENT ROW, not the end of the partition. Always add ROWS BETWEEN UNBOUNDED PRECEDING AND UNBOUNDED FOLLOWING when you want the true last value.

SQL — JSONB operators reference
-- Sample data: {"name": "Alice", "address": {"city": "NYC"}, "tags": ["vip","active"]}

-- -> returns JSONB, ->> returns text
SELECT
  metadata->'name'           AS name_json,    -- JSONB: "Alice"
  metadata->>'name'          AS name_text,    -- TEXT:  Alice
  metadata->'address'->>'city' AS city,          -- nested: NYC
  metadata#>>'{address,city}'  AS city_path,    -- path syntax
  metadata->'tags'->>0       AS first_tag,    -- array index

  -- Check key existence
  metadata ? 'premium'         AS has_premium,  -- key exists?
  metadata ?| ARRAY['vip', 'admin'] AS any_match,
  metadata ?& ARRAY['name', 'email'] AS all_present,

  -- Containment: does JSONB contain this subset?
  metadata @> '{"name": "Alice"}' AS is_alice,

  -- JSONB_ARRAY_ELEMENTS: expand array to rows
  JSONB_ARRAY_LENGTH(metadata->'tags') AS tag_count

FROM users;

-- Filter by JSONB content (uses GIN index)
SELECT * FROM users
WHERE metadata @> '{"status": "active", "plan": "enterprise"}';

Transactions & Concurrency
Isolation levels, advisory locks, SELECT FOR UPDATE, savepoints, and patterns for concurrent safe operations.
🔐 SELECT FOR UPDATE — Pessimistic Lockingessential
▼
SQL — Row-level locking
-- Standard: lock row for update (blocks other writers)
BEGIN;
  SELECT balance FROM accounts
  WHERE id = 42 FOR UPDATE;
  -- No one else can UPDATE this row until COMMIT/ROLLBACK
  UPDATE accounts SET balance = balance - 100 WHERE id = 42;
COMMIT;

-- FOR UPDATE SKIP LOCKED: queue processing pattern
BEGIN;
  SELECT * FROM job_queue
  WHERE status = 'pending'
  ORDER BY priority DESC
  LIMIT 1
  FOR UPDATE SKIP LOCKED;  -- skip rows locked by other workers
COMMIT;

-- FOR SHARE: shared lock (blocks writers, allows other readers)
SELECT * FROM users WHERE id = $1 FOR SHARE;

-- FOR NO KEY UPDATE: lighter lock (for FK checks)
SELECT * FROM accounts WHERE id = $1 FOR NO KEY UPDATE;

  ROLLBACK TO SAVEPOINT after_order;  -- only undo inventory change

🔑 Advisory Locks — Application-Level Lockingexpert
▼
Advisory locks are cooperative locks you define in your application — perfect for distributed cron jobs, singleton tasks, and feature gates.
SQL — Advisory lock patterns
-- Session-level advisory lock (held until released or disconnected)
SELECT pg_advisory_lock(12345);      -- blocks until acquired
-- ... do work ...
SELECT pg_advisory_unlock(12345);    -- release

-- Try-lock (non-blocking): returns false if already locked
DO $$
BEGIN
  IF pg_try_advisory_lock(42) THEN
    RAISE NOTICE 'Got lock, running job';
    -- run the cron job
    PERFORM pg_advisory_unlock(42);
  ELSE
    RAISE NOTICE 'Job already running, skipping';
  END IF;
END$$;

-- Transaction-level advisory lock (auto-released at commit/rollback)
BEGIN;
  SELECT pg_advisory_xact_lock(hashtext('import_job_user_' || $1));
  -- Locked for duration of transaction
COMMIT;

SELECT * FROM pg_locks WHERE locktype = 'advisory';

Pagination Patterns
The right way to paginate large datasets — from classic OFFSET to high-performance keyset pagination.
📄 OFFSET vs Keyset Paginationessential
▼
SQL — Pagination comparison
-- ❌ OFFSET pagination: slows exponentially at depth
-- Page 1000 = scan and discard 99,000 rows!
SELECT * FROM orders
ORDER BY created_at DESC
LIMIT 20 OFFSET 19980;  -- page 1000

-- ✅ Keyset (cursor) pagination: O(1) at any depth
-- First page:
SELECT id, created_at, total FROM orders
ORDER BY created_at DESC, id DESC
LIMIT 20;

-- Next page (pass last row's values as cursor)
SELECT id, created_at, total FROM orders
WHERE (created_at, id) < ('2024-11-15 10:30:00', 9876)
ORDER BY created_at DESC, id DESC
LIMIT 20;

-- Keyset with ties: handle same timestamp gracefully
SELECT id, score, name FROM leaderboard
WHERE (score, id) < (9500, 1234)  -- cursor tuple
ORDER BY score DESC, id DESC
LIMIT 20;

-- Bidirectional keyset with total count
WITH page_data AS (
  SELECT *, COUNT(*) OVER() AS total_count
  FROM orders
  WHERE user_id = $1
  ORDER BY created_at DESC
  LIMIT 20
)
SELECT *, total_count FROM page_data;
⚠️
OFFSET is fine for small datasets (<10k rows) or admin UIs. For user-facing APIs with large tables, always use keyset pagination — it's stable, fast at any depth, and index-friendly.

SQL — Cohort analysis
WITH
cohorts AS (
  SELECT
    user_id,
    DATE_TRUNC('month', MIN(created_at)) AS cohort_month
  FROM orders
  GROUP BY user_id
),
user_activity AS (
  SELECT
    o.user_id,
    c.cohort_month,
    DATE_TRUNC('month', o.created_at) AS activity_month,
    DATE_PART('month',
      AGE(DATE_TRUNC('month', o.created_at), c.cohort_month)
    ) AS months_since_cohort
  FROM orders o
  JOIN cohorts c USING(user_id)
)
SELECT
  cohort_month,
  months_since_cohort,
  COUNT(DISTINCT user_id) AS active_users,
  ROUND(100.0 * COUNT(DISTINCT user_id) /
    FIRST_VALUE(COUNT(DISTINCT user_id)) OVER (
      PARTITION BY cohort_month ORDER BY months_since_cohort
    ), 1
  ) AS retention_pct
FROM user_activity
GROUP BY cohort_month, months_since_cohort
ORDER BY cohort_month, months_since_cohort;

-- Generate complete date series (no gaps)
WITH date_series AS (
  SELECT d::date AS day
  FROM generate_series(

ORDER BY ds.day;

      product_id::text,
      quarter::text,
      SUM(revenue)::numeric
    FROM quarterly_sales
    GROUP BY 1, 2
    ORDER BY 1, 2
  $$,
  -- Categories (column names)
  $$VALUES ('Q1'), ('Q2'), ('Q3'), ('Q4')$$
) AS ct(product_id text, q1 numeric, q2 numeric, q3 numeric, q4 numeric);

-- Unpivot (columns → rows) using UNION ALL
SELECT id, 'Q1' AS quarter, q1_revenue AS revenue FROM annual_sales
UNION ALL
SELECT id, 'Q2', q2_revenue FROM annual_sales
UNION ALL
SELECT id, 'Q3', q3_revenue FROM annual_sales
UNION ALL
SELECT id, 'Q4', q4_revenue FROM annual_sales
ORDER BY id, quarter;

-- PL/pgSQL with error handling
CREATE OR REPLACE FUNCTION transfer_funds(
  from_account BIGINT,
  to_account   BIGINT,
  amount       NUMERIC
) RETURNS BOOLEAN LANGUAGE plpgsql AS $$
DECLARE
  from_balance NUMERIC;
BEGIN
  SELECT balance INTO STRICT from_balance
  FROM accounts WHERE id = from_account FOR UPDATE;

  IF from_balance < amount THEN
    RAISE EXCEPTION 'Insufficient funds: have %, need %',
      from_balance, amount
      USING ERRCODE = '23514';
  END IF;

  UPDATE accounts SET balance = balance - amount WHERE id = from_account;
  UPDATE accounts SET balance = balance + amount WHERE id = to_account;

  INSERT INTO transfer_log (from_id, to_id, amount, ts)
  VALUES (from_account, to_account, amount, now());

  RETURN TRUE;
EXCEPTION WHEN OTHERS THEN
  RAISE WARNING 'Transfer failed: %', SQLERRM;
  RETURN FALSE;
END$$;

-- Stored PROCEDURE (can commit/rollback, no return)
CREATE OR REPLACE PROCEDURE archive_old_orders(cutoff_days INT DEFAULT 365)
LANGUAGE plpgsql AS $$
DECLARE
  batch_size   INT := 5000;
  deleted_rows INT;
BEGIN
  LOOP
    WITH to_archive AS (
      DELETE FROM orders
      WHERE id IN (
        SELECT id FROM orders
        WHERE created_at < now() - (cutoff_days || ' days')::interval
          AND status = 'delivered'
        LIMIT batch_size
      )
      RETURNING *
    )
    INSERT INTO order_archive SELECT *, now() AS archived_at FROM to_archive;

    GET DIAGNOSTICS deleted_rows = ROW_COUNT;
    COMMIT;  -- procedures can commit mid-execution
    EXIT WHEN deleted_rows < batch_size;
  END LOOP;
END$$;

-- Call procedure
CALL archive_old_orders(730); -- 2-year cutoff

📋 Audit Log Triggeradvanced
▼
SQL — Full audit trail
CREATE TABLE audit_log (
  id          BIGSERIAL PRIMARY KEY,
  table_name  TEXT,
  operation   TEXT,  -- INSERT/UPDATE/DELETE
  old_data    JSONB,
  new_data    JSONB,
  changed_by  TEXT DEFAULT current_user,
  changed_at  TIMESTAMPTZ DEFAULT now()
);

CREATE OR REPLACE FUNCTION audit_trigger_fn()
RETURNS TRIGGER LANGUAGE plpgsql AS $$
BEGIN
  INSERT INTO audit_log (table_name, operation, old_data, new_data)
  VALUES (
    TG_TABLE_NAME,
    TG_OP,
    CASE WHEN TG_OP = 'INSERT' THEN NULL ELSE to_jsonb(OLD) END,
    CASE WHEN TG_OP = 'DELETE' THEN NULL ELSE to_jsonb(NEW) END
  );
  RETURN CASE WHEN TG_OP = 'DELETE' THEN OLD ELSE NEW END;
END$$;

-- Apply to sensitive tables
CREATE TRIGGER audit_users
  AFTER INSERT OR UPDATE OR DELETE ON users
  FOR EACH ROW EXECUTE FUNCTION audit_trigger_fn();

-- Query audit trail for a specific record
SELECT
  operation, changed_by, changed_at,
  old_data->>'email' AS old_email,
  new_data->>'email' AS new_email
FROM audit_log
WHERE table_name = 'users'
  AND (old_data->>'id' = '42' OR new_data->>'id' = '42')
ORDER BY changed_at DESC;

-- Array creation and access
SELECT
  ARRAY[1,2,3],                           -- literal array
  ARRAY[1,2,3][1],                        -- index (1-based!): 1

  ARRAY_APPEND(ARRAY[1,2], 3),            -- {1,2,3}

  ARRAY[1,2] <@ ARRAY[1,2,3];           -- contained in: true

WHERE tags && ARRAY['sql', 'nosql', 'database'];

SELECT post_id, unnest(tags) AS tag FROM posts;

SELECT ARRAY_AGG(DISTINCT tag ORDER BY tag) FROM all_tags;

SQL — Range types
-- Table with daterange for subscription periods
CREATE TABLE subscriptions (
  id          BIGSERIAL PRIMARY KEY,
  user_id     BIGINT,
  period      DATERANGE,       -- [start, end)
  EXCLUDE USING GIST (user_id WITH =, period WITH &&)
  -- ^ constraint: no overlapping periods per user
);

INSERT INTO subscriptions (user_id, period)
VALUES (1, '[2024-01-01, 2024-12-31)');

-- Overlap check (&&)
SELECT * FROM subscriptions
WHERE period && '[2024-06-01, 2024-09-01)'::daterange;

-- Containment (@>)
SELECT * FROM subscriptions
WHERE period @> '2024-07-15'::date;

-- Integer range: row number ranges, price tiers
SELECT int4range(1, 100) @> 50;  -- true: 50 is in [1,100)

-- tsrange for timestamp ranges
SELECT * FROM meetings
WHERE tstzrange(start_time, end_time) &&
      tstzrange('2024-11-15 09:00', '2024-11-15 17:00');

Subqueries
Scalar subqueries, correlated subqueries, EXISTS, and when subqueries beat joins.
🎯 Correlated vs Non-Correlated Subqueriesessential
▼
SQL
-- Non-correlated: runs once
SELECT * FROM orders
WHERE total > (SELECT AVG(total) FROM orders);

-- Correlated: runs once per outer row (potentially slow)
SELECT
  c.id, c.name,
  (SELECT COUNT(*) FROM orders o WHERE o.customer_id = c.id) AS order_count
FROM customers c;
-- Better: use LEFT JOIN + GROUP BY for large tables

-- Scalar subquery in SELECT (useful for single values)
SELECT
  o.id,
  o.total,
  o.total / (SELECT AVG(total) FROM orders) AS vs_avg_ratio
FROM orders o;

-- IN vs EXISTS: EXISTS short-circuits, faster for large sets
-- Use EXISTS when you only need to know IF a match exists
SELECT * FROM customers c
WHERE EXISTS (
  SELECT 1 FROM orders o
  WHERE o.customer_id = c.id
    AND o.total > 1000
);

-- ANY / ALL operators
SELECT * FROM products
WHERE price > ALL(SELECT price FROM competitor_products);

SELECT * FROM products
WHERE price > ANY(SELECT price FROM competitor_products);
